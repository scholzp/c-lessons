\begin{frame}[fragile ]{gcc for Unix-based operating systems}
	You maybe have to update your system before the installation\verb|!| \\
	Ubuntu / Debian:
	\begin{lstlisting}[numbers=none]
$ sudo apt-get update
$ sudo apt-get install gcc
    \end{lstlisting}
	\bigskip
	Arch Linux:
	\begin{lstlisting}[numbers=none]
$ sudo pacman -Syu
$ sudo pacman -S gcc
    \end{lstlisting}
	\bigskip
	Mac OS X:
	\begin{lstlisting}[numbers=none]
$ brew install gcc
    \end{lstlisting}
	\bigskip
	... and you're done ;-)
\end{frame}
